rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Demo rule cho giai doan phat trien.
    // Khi nop san pham that, nen dung Firebase Auth hoac chi cho Node-RED Admin SDK ghi du lieu.

    match /devices/{deviceId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /sensor_logs/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /rain_events/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /rack_events/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /alerts/{docId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    match /accounts/{accountId} {
      allow read, write: if request.auth != null;
    }
  }
}
